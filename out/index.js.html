const fs = require("fs");
const smartapi = require("smartapi-parser");
const jsnx = require('jsnetworkx');

/**
 * A knowledge graph representing how APIs are connected
 */
class KnowledgeGraph {
    /**
     * constructor description
     */
    constructor() {
        this.G = new jsnx.MultiDiGraph();
        this.api_list = this.listAllSpecs();
        this.loadSmartAPIs();
    }

    /**
     * Load All SmartAPI Specifications into memory
     * @param {string} source - the source of SmartAPI specification, default is 'local', could be specified as 'api'.
     * @return {Array} An array of objects, with each object representing one SmartAPI Specification
     */
    loadAllSpecs(source = 'local') {
        if (source === 'local') {
            return fs.readdirSync('./smartapi_specs');
        }
    }

    /**
     * Construct API Meta Knowledge Graph based on SmartAPI Specifications
     */
    constructMetaKnowledgeGraph() {
        let spec_path, api, ops;

        for (let spec_name of this.api_list) {
            spec_path = './smartapi_specs/' + spec_name;
            api = new smartapi.API(spec_path);
            ops = api.fetchAllOpts();
            op.map(op =>
                G.addEdge(
                    op.association.input_id,
                    op.association.output_id,
                    op
                )
            )
        }
    }
}


/**
 * Construct API Meta Knowledge Graph based on SmartAPI Specifications.
 * @param {Array} specs - list of smartapi specifications.
 * @return {MultiDiGraph} a networkx MultiDiGraph representing API connections.
 */
constructMetaKnowledgeGraph = (specs) => {
    let api, ops;
    let G = new jsnx.MultiDiGraph();
    specs.map(spec => {
        api = new smartapi.API(spec);
        ops = api.fetchAllOpts();
        op.map(op =>
            G.addEdge(
                op.association.input_id,
                op.association.output_id,
                op
            )
        )
    })
    return G
}

exports.KnowledgeGraph = KnowledgeGraph;
